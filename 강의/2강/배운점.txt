1. Array.from 새로운 배열을 만든다
Array.from({length:5}, (v.i) => i);     // [1,2,3,4,5]
v : 값
i : 인덱스
ex) --
Array.from([5,4,3,2,1],(v,i) => v)      // [5,4,3,2,1]
Array.from({length:5}, () => 1);        // [1,1,1,1,1]
